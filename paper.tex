\documentclass{article}
\usepackage{ocgx2}

% Define a command for creating a collapsible section
\newcommand{\collapsiblesection}[2]{%
  \begin{ocg}{#2}{#2}{off}#1\end{ocg}%
  \textLayer{\textattachfile[color=0 0 1]{#2}{\textbf{Toggle #2}}}%
}

\begin{document}

\section{Introduction}
This is the introductory section of your paper.

\collapsiblesection{%
  \section{Theorem 1}
  Here is the statement of Theorem 1.
}{theorem1}

\collapsiblesection{%
  \subsection{Proof of Theorem 1}
  Here is the proof for Theorem 1.
}{proof1}

\collapsiblesection{%
  \section{Theorem 2}
  Here is the statement of Theorem 2.
}{theorem2}

\collapsiblesection{%
  \subsection{Proof of Theorem 2}
  Here is the proof for Theorem 2.
}{proof2}

\section{Conclusion}
This is the conclusion of your paper.

\end{document}
